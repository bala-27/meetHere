function arrayUtil() {
    /**
     * Finds the index of a nested object in an array
     *
     * @extends Array
     */
    Array.prototype.deepIndexOf = function (value) {
        for (var i = 0; i < this.length; ++i) {
            if (this[i].length === value.length &&
                this[i].every(function (v, j) { return v === value[j]; })) {
                return i;
            }
        }
        return -1;
    };
}
